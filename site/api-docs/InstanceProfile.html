def initialize(hwp_name, args = {})
  opts = args.inject({ :id => hwp_name.to_s }) do |m, e|
    k, v = e
    m[$1] = v if k.to_s =~ /^hwp_(.*)$/
    m
  end
  super(opts)
end

def overrides
  [:memory, :storage, :architecture, :cpu].inject({}) do |h, p|
    if v = instance_variable_get("@#{p}")
      h[p] = v
    end
    h
  end
end
