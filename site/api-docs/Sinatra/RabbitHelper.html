def entry_points
  collections.values.inject([]) do |m, coll|
    url = url_for coll.operations[:index].path, :full
    m << [ coll.name, url ]
  end
end

def query_url(url, params)
  return url if params.nil? || params.empty?
  url + "?#{URI.escape(params.collect{|k,v| "#{k}=#{v}"}.join('&'))}"
end
